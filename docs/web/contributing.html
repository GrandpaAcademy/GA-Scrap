Getting Started

1. Fork and Clone

# Fork the repository on GitHub, then clone your fork
git clone https://github.com/YOUR_USERNAME/GA-Scrap.git
cd GA-Scrap

# Add upstream remote
git remote add upstream https://github.com/GrandpaAcademy/GA-Scrap.git

2. Create a Branch

# Create a new branch for your contribution
git checkout -b feature/your-feature-name

# Or for bug fixes
git checkout -b fix/issue-description

3. Make Your Changes

# Make your changes and commit them
git add .
git commit -m "Add: descriptive commit message"

# Push to your fork
git push origin feature/your-feature-name

4. Create Pull Request

Go to GitHub and create a pull request from your branch to the main repository.

Code Guidelines

Python Style
  • Follow PEP 8 style guide
  • Use type hints for function parameters and return values
  • Write docstrings for all public methods
  • Keep line length under 88 characters
  • Use meaningful variable names
Code Quality
  • Write unit tests for new features
  • Ensure backward compatibility
  • Handle errors gracefully
  • Add logging where appropriate
  • Keep methods focused and small
Example Code Style
from typing import Optional, List, Dict, Any

class SyncGAScrap:
    """Synchronous web scraping interface for GA-Scrap.
    
    This class provides a simple, synchronous API for web scraping
    operations using Playwright under the hood.
    
    Args:
        sandbox_mode: Enable error-resilient sandbox mode
        headless: Run browser in headless mode
        timeout: Default timeout in milliseconds
    """
    
    def __init__(
        self, 
        sandbox_mode: bool = False,
        headless: bool = True,
        timeout: int = 30000
    ) -> None:
        self.sandbox_mode = sandbox_mode
        self.headless = headless
        self.timeout = timeout
        self._page: Optional[Any] = None
    
    def get_text(self, selector: str, timeout: Optional[int] = None) -> Optional[str]:
        """Extract text content from an element.
        
        Args:
            selector: CSS selector or XPath expression
            timeout: Override default timeout
            
        Returns:
            Text content of the element, or None if not found
            
        Example:
            >>> scraper.get_text("h1")
            "Welcome to Example.com"
        """
        try:
            element = self._page.locator(selector)
            return element.text_content(timeout=timeout or self.timeout)
        except Exception as e:
            if self.sandbox_mode:
                self._log_error(f"Failed to get text for {selector}: {e}")
                return None
            raise

Testing

Writing Tests

# tests/test_sync_scraper.py
import pytest
from ga_scrap import SyncGAScrap

class TestSyncGAScrap:
    """Test suite for SyncGAScrap class."""
    
    def test_basic_navigation(self):
        """Test basic page navigation."""
        with SyncGAScrap(headless=True) as scraper:
            scraper.goto("https://httpbin.org/html")
            title = scraper.get_text("h1")
            assert title is not None
            assert "Herman Melville" in title
    
    def test_sandbox_mode_error_handling(self):
        """Test that sandbox mode handles errors gracefully."""
        with SyncGAScrap(sandbox_mode=True, headless=True) as scraper:
            scraper.goto("https://httpbin.org/html")
            
            # This should not crash in sandbox mode
            result = scraper.click("#non-existent-element")
            assert result is None
            
            # This should still work
            title = scraper.get_text("h1")
            assert title is not None
    
    @pytest.mark.parametrize("browser_type", ["chromium", "firefox", "webkit"])
    def test_multiple_browsers(self, browser_type):
        """Test functionality across different browsers."""
        with SyncGAScrap(browser_type=browser_type, headless=True) as scraper:
            scraper.goto("https://httpbin.org/html")
            title = scraper.get_text("h1")
            assert title is not None
