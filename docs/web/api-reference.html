Complete Method Documentation

Every method, parameter, and return value

Overview

GA-Scrap provides a comprehensive API that covers all web scraping needs. All methods are available in both synchronous and asynchronous interfaces.

Convention: This reference shows the synchronous interface. For async, simply add await before method calls and use GAScrap instead of SyncGAScrap.

Constructor

SyncGAScrap()

Creates a new GA-Scrap instance with the specified configuration.

Parameters:
Parameter Type Default Description
sandbox_mode bool False Enable error-resilient sandbox mode
headless bool True Run browser in headless mode
browser_type str "chromium" Browser engine: "chromium", "firefox", "webkit"
timeout int 30000 Default timeout in milliseconds
viewport dict {"width": 1280, "height": 720} Browser viewport size
user_agent str None Custom user agent string
device str None Device to emulate (e.g., "iPhone 12")
debug bool False Enable detailed logging
Example:
from ga_scrap import SyncGAScrap

# Basic usage
scraper = SyncGAScrap()

# Development configuration
scraper = SyncGAScrap(
    sandbox_mode=True,
    headless=False,
    debug=True,
    timeout=60000
)

# Mobile device emulation
scraper = SyncGAScrap(
    device="iPhone 12",
    sandbox_mode=True
)

Lifecycle Methods

start()

Initialize browser and create context.

scraper = SyncGAScrap()
scraper.start()
# Use scraper methods
scraper.stop()
stop()

Close browser and cleanup resources.

# Context manager (recommended)
with SyncGAScrap() as scraper:
    # Automatically calls start() and stop()
    pass

Navigation Methods

Navigate to a URL

Parameters:

  • url (str): Target URL
  • wait_until (str): When to consider navigation complete ("load", "domcontentloaded", "networkidle")
  • timeout (int): Navigation timeout in milliseconds
scraper.goto("https://example.com")
scraper.goto("https://example.com", wait_until="networkidle")
scraper.goto("https://example.com", timeout=60000)

Browser navigation controls

scraper.go_back()     # Navigate back in history
scraper.go_forward()  # Navigate forward in history
scraper.reload()      # Reload current page

Element Interaction

Method Parameters Description Example
click(selector) selector, button="left", timeout=None Click an element scraper.click("button")
click_text(text) text, timeout=None Click element by text content scraper.click_text("Submit")
double_click(selector) selector, timeout=None Double click an element scraper.double_click(".item")
right_click(selector) selector, timeout=None Right click an element scraper.right_click(".menu")
input(selector, text) selector, text, timeout=None Type text into input field scraper.input("#username", "john")
clear_input(selector) selector, timeout=None Clear input field scraper.clear_input("#password")
select_option(selector, value) selector, value, timeout=None Select dropdown option scraper.select_option("select", "option1")
check(selector) selector, timeout=None Check checkbox/radio button scraper.check("#terms")
uncheck(selector) selector, timeout=None Uncheck checkbox scraper.uncheck("#newsletter")
hover(selector) selector, timeout=None Hover over element scraper.hover(".dropdown")
focus(selector) selector, timeout=None Focus on element scraper.focus("input")

Data Extraction

Text Extraction
# Single element
title = scraper.get_text("h1")
price = scraper.get_text(".price")

# Multiple elements
items = scraper.get_all_text(".item")
links = scraper.get_all_text("a")
Attribute Extraction
# Single attribute
href = scraper.get_attribute("a", "href")
src = scraper.get_attribute("img", "src")

# Multiple attributes
urls = scraper.get_all_attributes("a", "href")
images = scraper.get_all_attributes("img", "src")
HTML Extraction
# Outer HTML
html = scraper.get_html(".content")

# Inner HTML
inner = scraper.get_inner_html(".container")
Element Information
# Check if element exists
exists = scraper.element_exists(".popup")

# Get element count
count = scraper.count_elements(".item")

# Get element properties
visible = scraper.is_visible(".modal")
enabled = scraper.is_enabled("button")

Waiting & Timing

Method Parameters Description
wait(milliseconds) milliseconds (int) Wait for specified time
wait_for_selector(selector) selector, timeout=None, state="visible" Wait for element to appear
wait_for_text(text) text, timeout=None Wait for text to appear on page
wait_for_url(pattern) pattern, timeout=None Wait for URL to match pattern
wait_for_load_state(state) state="load", timeout=None Wait for page load state
wait_for_network_idle() timeout=None Wait for network to be idle
Examples:
# Wait for element
scraper.wait_for_selector(".loading", state="hidden")  # Wait for loading to disappear
scraper.wait_for_selector(".results")  # Wait for results to appear

# Wait for navigation
scraper.wait_for_url("**/success")  # Wait for success page
scraper.wait_for_load_state("networkidle")  # Wait for network idle

# Wait for content
scraper.wait_for_text("Welcome")  # Wait for welcome message

Media & Files

Screenshots
# Basic screenshot
scraper.screenshot("page.png")

# Full page screenshot
scraper.screenshot_full_page("full.png")

# Element screenshot
scraper.screenshot_element(".chart", "chart.png")

# Custom options
scraper.screenshot("page.png", 
                  quality=90, 
                  type="jpeg")
File Operations
# Upload files
scraper.upload_files("input[type=file]", 
                     ["file1.txt", "file2.pdf"])

# Download files
scraper.download_file("https://example.com/file.zip")

# PDF generation
scraper.save_pdf("page.pdf")

# Set download directory
scraper.set_download_path("./downloads")

Page Manipulation

Scrolling
# Scroll to element
scraper.scroll_to_element(".footer")

# Scroll by amount
scraper.scroll_by(0, 500)  # x, y pixels

# Scroll to position
scraper.scroll_to_position(0, 1000)

# Scroll to bottom/top
scraper.scroll_to_bottom()
scraper.scroll_to_top()
Page Control
# Viewport control
scraper.set_viewport_size(1920, 1080)
size = scraper.get_viewport_size()

# Page information
url = scraper.get_current_url()
title = scraper.get_page_title()

# Multi-tab support
page2 = scraper.new_page()
scraper.goto("https://example.com", page=page2)
scraper.close_page(page2)

Network & Cookies

Cookie Management
# Get cookies
cookies = scraper.get_cookies()

# Add cookies
scraper.add_cookies([{
    "name": "session",
    "value": "abc123",
    "domain": "example.com"
}])

# Clear cookies
scraper.clear_cookies()
Network Control
# Block requests
scraper.block_requests(["*.png", "*.jpg"])

# Set user agent
scraper.set_user_agent("Custom Agent")

# Offline mode
scraper.set_offline(True)

# Network monitoring
requests = scraper.get_network_requests()

JavaScript Execution

# Execute JavaScript
result = scraper.execute_script("return document.title")
scraper.execute_script("document.body.style.background = 'red'")

# Execute with arguments
result = scraper.execute_script("return arguments[0] + arguments[1]", 5, 10)

# Inject CSS
scraper.inject_css("body { font-size: 20px; }")

# Add script tag
scraper.add_script_tag(url="https://cdn.jsdelivr.net/npm/jquery@3.6.0/dist/jquery.min.js")

Direct Playwright Access

# Get Playwright objects
page = scraper.get_playwright_page()
context = scraper.get_playwright_context()
browser = scraper.get_playwright_browser()

# Safe method execution
title = scraper.execute_playwright_method('page', 'title')
cookies = scraper.execute_playwright_method('context', 'cookies')

# Helper methods
title = scraper.playwright_page_method('title')
cookies = scraper.playwright_context_method('cookies')

# Use any Playwright feature
await page.evaluate("console.log('Direct Playwright access')")
await context.set_geolocation({"latitude": 40.7128, "longitude": -74.0060})

Method Chaining

Most methods support chaining for fluent, readable code:

# Chain multiple operations
result = (scraper
          .goto("https://example.com")
          .wait_for_selector(".content")
          .scroll_to_bottom()
          .click(".load-more")
          .wait_for_selector(".new-content")
          .get_all_text(".item"))

Error Handling

Sandbox Mode
# Enable sandbox mode for error resilience
with SyncGAScrap(sandbox_mode=True) as scraper:
    scraper.goto("https://example.com")
    
    # These won't crash if they fail
    scraper.click("#might-not-exist")
    scraper.input("#fake-input", "test")
    
    # This will still work
    title = scraper.get_text("h1")
Manual Error Handling
# Traditional try/catch
try:
    scraper.click("#button")
except Exception as e:
    print(f"Click failed: {e}")

# Check if element exists first
if scraper.element_exists("#button"):
    scraper.click("#button")
else:
    print("Button not found")
